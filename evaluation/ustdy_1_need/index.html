def load_data(
	participant_id: str,
	task_id: str,
	condition_id: str,
	columns: list[str]
) -> pd.DataFrame:
	# file names have the pattern XXX_YY_Z.csv

	# load the recording
	df = pd.read_csv(file_name).set_index("timestamp_ms").sort_index()
